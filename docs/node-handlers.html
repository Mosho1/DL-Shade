        Array: function (node) {
        var list = "{";
        if (node.elements) {
            _.each(node.elements, function (element) {
                list += this.compileNode(element) + ",";
            }, this);
        }
        list = list.substring(0, list.length - 1) + "}";
        return list;
    },
